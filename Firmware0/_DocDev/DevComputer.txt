[ ] Install Arduino from K:\TheKit\Arduino
[ ] Tool "Arduino"
    - Menu "File", select "Preference"
        - Box "Additional Boards Manager URLS", enter
          "https://raw.githubusercontent.com/espressif/arduino-esp32/gh-pages/package_esp32_index.json"
        - Click "OK"
    - Menu "Tools", select "Board" and select "Board Manager..."
        - Find "esp32" and click "Install"
    - Menu "Tools", select "Board", select "ESP32 Arduino" and select
      "OLIMEX ESP32-PoE-ISO"
